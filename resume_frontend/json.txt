import React, { useEffect, useState } from 'react'
import { FaBrain, FaPaperPlane } from 'react-icons/fa'


import { generateResume } from '../api/ResumeService';
import toast from 'react-hot-toast';
import { FaTrashCan } from 'react-icons/fa6';
import { BiBadge, BiBook, BiBookAdd, BiBookBookmark, BiColor } from 'react-icons/bi';

function GenerateResume() {

  // const [count, setCount] = new useState(0);
  const [description, setDescription] = new useState("");
  const [loading, setLoading] = new useState(false);

  // createt the state for the persolnal information
  const [data, setData] = useState({
    "personalInformation": {
      "fullName": "John Doe",
      "email": "john.doe@example.com",
      "phoneNumber": "+1234567890",
      "location": "New York, NY",
      "linkedIn": "https://www.linkedin.com/in/johndoe",
      "gitHub": "https://github.com/johndoe",
      "portfolio": "https://johndoe.com"
    },
  });


  const handleGenerate = async () => {
    console.log("handle Gengerate are called ...")
    console.log(description);

    try {
      setLoading(true)
      const responseData = await generateResume(description);
      console.log(responseData);

      toast.success("Resume Generated Loading", {
        position: "top-center",
        duration: 3000,
        closeOnClick: true,
        theme: "dark",
      })

    } catch (error) {
      console.log(error);

      toast.error("Resume Not Generated !", {
        position: "top-center",
        duration: 3000,
        theme: "dark"
      })
    } finally {
      setLoading(false);
      setDescription("");
    }
  }

  const handleClear = () => {
    setDescription("");
  }
  function showInputField() {
    return (<div className='bg-base-200 shawdow-lg rounded-lg p-10 max-w-2xl text-center border-white '>

      <h1 className='font-bold text-white text-4xl mb-6 flex justify-center items-center gap-2'>
        <FaBrain className='text-accent' />Ai Resume Description Input
      </h1>
      <p className='md-4 text-lg text-gray-600'>Enter the Detailed Description about yourself to generate your profeccional resume </p>

      <textarea className='textarea textarea-accent w-full  h-48 mb-6 resize-none mt-3'
        disabled={loading}

        placeholder='Type your Description Here'
        value={description}
        onChange={(e) => setDescription(e.target.value)}

      >
      </textarea>

      <div className='flex justify-center items-center gap-4 '>
        <button className='btn btn-primary m-2'

          disabled={loading}
          onClick={handleGenerate}
        >
          {loading && <span class="loading loading-spinner text-primary"></span>}

          <FaPaperPlane />
          Generate Resume

        </button>
        <button className='btn btn-secondary flex items-center gap-2 ml-3'
          onClick={handleClear}>
          <FaTrashCan />
          Clear
        </button>
      </div>

    </div>);
  }


  // create a function 
  function handleChange(event) {

    console.log("Event : " + event.target.value)

    console.log("Event : " + event.target.value)

    // console.log(...data);

    // when we simply give the information to the setData directly then input box not chnages. 
    setData({
      ...data,
      // Changing Personal Information
      personalInformation: {
        ...data.personalInformation,
        [event.target.name]: event.target.value,

      }
    })

  }

  useEffect(() => {
    console.log(data);
  }, [data])

  function showForm() {
    return (
      <div className='w-full'>

        <h1 className='font-bold text-white text-4xl mb-6 flex justify-center items-center gap-2'>
          <BiBook className='text-accent' />Resume Form
        </h1>

        <div className='w-full p-10'>
          <h1 className='font-bold'>Personal Information</h1>

          <div className='grid grid-cols-12 justify-between gap-5 p-4'>

            <div className='col-span-12 lg:col-span-6'>
              <label htmlFor="name" className='w-10'>FullName : </label>
              <input type='text'

                onChange={handleChange}
                name='fullName'
                placeholder='Enter Name'
                value={data.personalInformation.fullName}
                id='name'
                className='input input-bordered w-full my-5'
              />

            </div>

            <div className='col-span-12 lg:col-span-6'>
              <label htmlFor='email' >Email :</label>
              <input type='email'
                onChange={handleChange}
                name='email'
                placeholder='Enter Email'
                value={data.personalInformation.email}
                className='input input-bordered w-full my-5 font-bold'
                id='email'
              />


            </div>

            <div className='col-span-12 lg:col-span-6 '>
              <label htmlFor='number' >Phone Number :</label>
              <input type='text'
                onChange={handleChange}
                name='phoneNumber'
                placeholder='Enter Number'
                value={data.personalInformation.phoneNumber}
                className='input input-bordered w-full my-5 font-bold'
                id='number'
              />


            </div>

            <div className='col-span-12 lg:col-span-6 '>
              <label htmlFor='location' >Location :</label>
              <input type='text'
                onChange={handleChange}
                name='location'
                placeholder='Enter Location'
                value={data.personalInformation.location}
                className='input input-bordered w-full my-5 font-bold'
                id='location'
              />


            </div>

            <div className='col-span-12 lg:col-span-6 '>
              <label htmlFor='hithub' >Github :</label>
              <input type='text'
                onChange={handleChange}
                name='github'
                placeholder='Enter Github Link'
                value={data.personalInformation.gitHub}
                className='input input-bordered w-full my-5 font-bold'
                id='github'
              />


            </div>

            <div className='col-span-12 lg:col-span-6 '>
              <label htmlFor='linkedin' >LinkedIn :</label>
              <input type='text'
                onChange={handleChange}
                name='linkedIn'
                placeholder='Enter LinkedIn link'
                value={data.personalInformation.linkedIn}
                className='input input-bordered w-full my-5 font-bold'
                id='linkedin'
              />


            </div>

            <div className='col-span-12 lg:col-span-12 '>
              <label htmlFor='portfoliol' className='text-gray-500 '  >Portfolio :</label>
              <input type='text'
                onChange={handleChange}
                name='portfolio'
                placeholder='Enter Portfolio Link'
                value={data.personalInformation.portfolio}
                className='input input-bordered w-full my-5 font-bold '
                id='portfolio'

              />
            </div>
          </div>
        </div>
      </div>
    )
  }

  // const description[description , setDescription] = ;
  return (
    <div className=' flex flex-col justify-center items-center font-sans gap-8'>
      {showForm()}
      {showInputField()}

    </div>
  )
}

export default GenerateResume
